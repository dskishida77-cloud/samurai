  <title>加算プログラム</title>

  <script>

    // 加算を行う関数

    function add(a, b) {

      return a + b;

    }


    // ボタンが押されたときの処理

    function calculate() {

      const num1 = Number(document.getElementById("num1").value);

      const num2 = Number(document.getElementById("num2").value);

      const result = add(num1, num2);

      document.getElementById("result").value = result;

    }

  </script>

</head>

<body>

  <h2>加算プログラム</h2>

  <p>

    数値1：<input type="number" id="num1">

  </p>

  <p>

    数値2：<input type="number" id="num2">

  </p>

  <p>

    <button onclick="calculate()">加算</button>

  </p>

  <p>

    結果：<input type="text" id="result" readonly>

  </p>
